```python
"""Numeric parsing helpers shared across pipeline modules."""
from __future__ import annotations

import logging
from typing import Any

from src.utils.logger import get_logger

logger = get_logger("utils.number_utils")


def safe_float(
    value: Any,
    *,
    field_name: str | None = None,
    strip_commas: bool = False,
    default: float = 0.0,
    log_level: int = logging.DEBUG,
) -> float:
    """Parse ``value`` as float with forgiving fallbacks.

    Args:
        value: Raw input (str, int, float, None, etc.).
        field_name: Optional label used in failure log for diagnostics.
        strip_commas: When True, remove ``,`` from string inputs before
            parsing (e.g. ``"1,234.5"``).
        default: Returned when parsing fails or input is None/empty.
        log_level: ``logging`` level used when a parse fails.

    Returns:
        Parsed float, or ``default`` on failure.
    """
    if value is None or value == "":
        return default
    try:
        if strip_commas:
            return float(str(value).replace(",", ""))
        return float(value)
    except (TypeError, ValueError):
        logger.log(
            log_level,
            "safe_float failed field=%s value=%r; defaulting to %s",
            field_name,
            value,
            default,
        )
        return default
```

    assert safe_float("") == 0.0
    assert safe_float(None, default=-1.0) == -1.0

def _staleness_threshold_seconds() -> int:
    """Read LISTENER_STALENESS_SECONDS with safe fallback.

## 8. 프롬프트 사용 가이드

- 각 프롬프트는 **한 세션에 하나씩**만 투입한다. 하나의 세션에서 여러 티켓을
  연속 실행하면 worktree CWD 이동 중 인지 오류로 다른 브랜치에 커밋될
  위험이 있다.
- 프롬프트의 "사전 확인" 섹션은 Claude Code가 **반드시 수행**해야 한다.
  이 확인을 스킵하면 `main`에 직접 커밋해 전체 계획이 붕괴할 수 있다.
- `git add -A` 사용 금지 원칙을 모든 프롬프트에 명시했다. 이는 다른 worktree와의
  mount 공유 환경(Cowork)에서 의도치 않은 파일 스테이징을 막기 위함.
- 각 프롬프트는 롤백 커맨드를 포함한다. 문제 발생 시 해당 worktree 단독으로
  되돌리고 다른 티켓에는 영향이 없다.
