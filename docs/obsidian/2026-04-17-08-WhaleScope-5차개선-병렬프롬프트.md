# WhaleScope 5차 개선 병렬 프롬프트

## STRICT_SCOPE 원칙

> **절대 규칙**: 각 트랙은 `ALLOWED_FILES`에 명시된 파일만 수정/생성할 수 있다.
> `FORBIDDEN_FILES`에 명시된 파일을 수정하면 머지 시 충돌이 발생하므로 해당 트랙은 즉시 reject한다.
> 다른 트랙의 파일을 import하거나 읽는 것은 허용하되, **수정은 금지**한다.

```typescript
// apps/dashboard/app/api/dashboard/route.ts
import { getCached, setCache, DASHBOARD_CACHE_TTL } from "@/lib/cache";
import { rateLimit, rateLimitResponse, API_RATE_LIMIT } from "@/lib/rate-limit";

export const runtime = "nodejs";
// force-dynamic 제거, 또는 revalidate 설정
export const revalidate = 60;

export async function GET(request: Request) {
  // Rate limit check
  const ip = request.headers.get("x-forwarded-for") || "unknown";
  const rl = rateLimit(ip, API_RATE_LIMIT);
  if (!rl.allowed) return rateLimitResponse(rl.retryAfter!);

  const unauthorized = requireDashboardAuth(request);
  if (unauthorized) return unauthorized;

  // Cache check
  const cached = getCached<DashboardData>("dashboard");
  if (cached) return NextResponse.json(cached);

  const data = await getDashboardData();
  setCache("dashboard", data, DASHBOARD_CACHE_TTL);
  return NextResponse.json(data);
}
```

```tsx
"use client";

import { useState } from "react";

type SignalActionCardProps = {
  signalId: string;
  rule: string;
  severity: string;
  summary: string;
  confidenceLabel: string;
  severityLabel: string;
  sourceLabel: string;
  tone: string;
  evidenceTxHashes: string[];
};

export function SignalActionCard(props: SignalActionCardProps) {
  const [dismissed, setDismissed] = useState(false);
  const [acknowledging, setAcknowledging] = useState(false);

  async function handleDismiss() {
    setAcknowledging(true);
    await fetch(`/api/signals/${props.signalId}`, {
      method: "PATCH",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ action: "dismiss" }),
    });
    setDismissed(true);
    setAcknowledging(false);
  }

  async function handleAcknowledge() {
    setAcknowledging(true);
    await fetch(`/api/signals/${props.signalId}`, {
      method: "PATCH",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ action: "acknowledge" }),
    });
    setAcknowledging(false);
  }

  if (dismissed) return null;

  return (
    <div className={`signal-card signal-card--${props.tone}`}>
      <div className="signal-card__header">
        <span className="signal-card__rule">{props.rule}</span>
        <span className="signal-card__severity">{props.severityLabel}</span>
      </div>
      <p className="signal-card__summary">{props.summary}</p>
      <div className="signal-card__meta">
        <span>{props.confidenceLabel}</span>
        <span>{props.sourceLabel}</span>
      </div>
      <div className="signal-card__actions">
        <button onClick={handleAcknowledge} disabled={acknowledging}>
          확인됨
        </button>
        <button onClick={handleDismiss} disabled={acknowledging}>
          무시
        </button>
      </div>
    </div>
  );
}
```

```typescript
// apps/dashboard/app/api/signals/[id]/route.ts
import { NextResponse } from "next/server";
import { requireDashboardAuth } from "@/lib/auth";
import { getSheets } from "@/lib/sheets";

export const runtime = "nodejs";

export async function PATCH(
  request: Request,
  { params }: { params: { id: string } }
) {
  const unauthorized = requireDashboardAuth(request);
  if (unauthorized) return unauthorized;

  const { action } = await request.json();
  if (!["dismiss", "acknowledge"].includes(action)) {
    return NextResponse.json({ error: "Invalid action" }, { status: 400 });
  }

  const sheets = getSheets();
  // signals 탭에서 signal_id 매칭 행에 status 열 업데이트
  await sheets.updateSignalStatus(params.id, action);

  return NextResponse.json({ ok: true, signalId: params.id, action });
}
```

```tsx
"use client";

import { useState, useEffect } from "react";

type WatchlistEntry = {
  address: string;
  chain: string;
  label: string;
  enabled: boolean;
};

export function WatchlistEditor() {
  const [entries, setEntries] = useState<WatchlistEntry[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    fetch("/api/watchlist")
      .then(r => r.json())
      .then(data => { setEntries(data.addresses || []); setLoading(false); });
  }, []);

  async function toggleAddress(address: string, enabled: boolean) {
    await fetch("/api/watchlist", {
      method: "PATCH",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ address, enabled }),
    });
    setEntries(prev =>
      prev.map(e => e.address === address ? { ...e, enabled } : e)
    );
  }

  if (loading) return <div>불러오는 중...</div>;

  return (
    <div className="watchlist-editor">
      <h3>감시 주소 관리</h3>
      {entries.map(entry => (
        <div key={entry.address} className="watchlist-entry">
          <span>{entry.label || entry.address.slice(0, 10) + "..."}</span>
          <span>{entry.chain}</span>
          <button onClick={() => toggleAddress(entry.address, !entry.enabled)}>
            {entry.enabled ? "비활성화" : "활성화"}
          </button>
        </div>
      ))}
    </div>
  );
}
```

```typescript
// apps/dashboard/app/api/watchlist/route.ts
import { NextResponse } from "next/server";
import { requireDashboardAuth } from "@/lib/auth";
import { getSheets } from "@/lib/sheets";

export const runtime = "nodejs";

export async function GET(request: Request) {
  const unauthorized = requireDashboardAuth(request);
  if (unauthorized) return unauthorized;

  const sheets = getSheets();
  const addresses = await sheets.listWatchedAddresses();
  return NextResponse.json({ addresses });
}

export async function PATCH(request: Request) {
  const unauthorized = requireDashboardAuth(request);
  if (unauthorized) return unauthorized;

  const { address, enabled } = await request.json();
  const sheets = getSheets();
  await sheets.toggleWatchedAddress(address, enabled);
  return NextResponse.json({ ok: true });
}
```
